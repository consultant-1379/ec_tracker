CREATE TABLE ecs
       (EC_ID        	  CHAR(150) NOT NULL PRIMARY KEY,
        SHIPMENT_NAME     CHAR(50) NOT NULL,
        RELEASE_NAME      CHAR(50) NOT NULL,
		PACKAGE_NAME      CHAR(150) NOT NULL,
		REQUESTER		  CHAR(20),
        EC_TYPE    		  CHAR(10) NOT NULL
						  CHECK(EC_TYPE IN ('MWS EC','HOT EC')),
		EU_NUMBER         CHAR(10),
		PACKAGE_GASKLINK  CHAR(200) NOT NULL,
		PACKAGE_TESTED 	  CHAR(10) NOT NULL
						  CHECK(PACKAGE_TESTED IN ('YES','NO')),
		README_LINK		  CHAR(200) NOT NULL,
		README_REVIEWED	  CHAR(10) NOT NULL
						  CHECK(README_REVIEWED IN ('YES','NO')),
		COMMENT			  CHAR(255),
		SIGNUM			  CHAR(150) NOT NULL,
		DELIVERED_TO_DM	  DATETIME,
		DELIVERED_TO_ST	  DATETIME,
		DELIVERED_TO_FOA  DATETIME)
;

CREATE TABLE trs
		(TR_ID			  INTEGER NOT NULL AUTO_INCREMENT PRIMARY KEY,
		TR_NUMBER     	  CHAR(20) NOT NULL,
		 SLOGAN			  CHAR(200) NOT NULL,
		 ORIGINATION	  CHAR(100) NOT NULL,
		 EC_ID  		  CHAR(150) NOT NULL)
		 ;

		(USER_ID		INTEGER NOT NULL AUTO_INCREMENT PRIMARY KEY,
		 USER_SIGNUM	CHAR(20) NOT NULL,
		 USER_ROLE		CHAR(20) NOT NULL
		);
